import os
import argparse

def main():
    parser = argparse.ArgumentParser(description='Ethical Tool on Kali')
    parser.add_argument('--option', type=str, help='Specify the option', required=True)
    
    args = parser.parse_args()
    
    if args.option == 'iso.ig':
        os.system('chmod +x main.py')
        print("Permissions changed for main.py")
    else:
        print("Invalid option. Please use 'iso.ig'.")

if __name__ == "__main__":
    main()
